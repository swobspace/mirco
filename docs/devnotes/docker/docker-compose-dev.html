docker-compose.dev.yml

docker-compose.dev.yml
version: "3.8"
# include:
#   - .docker/postgres.yml
#   - .docker/redis.yml

x-base: &base
  build:
    context: .
    dockerfile: .docker/Dockerfile.dev
  stdin_open: true
  tty: true

services:
  app:
    <<: *base
    command: bin/dev
    restart: on-failure
    volumes:
      - .:/app:Z,cached
    ports:
      - "3000:3000"
#     depends_on:
#       - postgres
#       - redis
    env_file:
      - .env.development.local
Table 1. Some Hints
Hint Description

external DB + Redis

I’m using PostgreSQL und Redis on my docker host for now.

--volume .:/app:Z,cached

Rootless containers on Fedora hosts don’t have access to the local working directory if selinux is enabled. The option Z takes care of it.
